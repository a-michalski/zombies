## ğŸ”§ INTEGRATION NOTES

After generating audio, integrate with `expo-av`:

```typescript
import { Audio } from 'expo-av';

// Load and play music
const [sound, setSound] = useState<Audio.Sound>();

async function playMusic() {
  const { sound } = await Audio.Sound.createAsync(
    require('@/assets/audio/music/gameplay.mp3'),
    { isLooping: true, volume: 0.6 }
  );
  setSound(sound);
  await sound.playAsync();
}

// Play SFX
async function playSFX(file: string) {
  const { sound } = await Audio.Sound.createAsync(
    require(`@/assets/audio/sfx/${file}.wav`)
  );
  await sound.playAsync();
  sound.setOnPlaybackStatusUpdate((status) => {
    if (status.didJustFinish) {
      sound.unloadAsync(); // Clean up
    }
  });
}
```
